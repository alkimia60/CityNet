- action = ContainerRegister.
    parameters:
        String token    token de sessió
        String id       identificador de 6 caracters alfanumèrics
        String type     ha de ser un d'aquests {"paper", "glass", "packaging", "organic", "trash"}
        double longitude   format p.e.: 41.38994598
        double latitude    format p.e.: 2.22962594
    return:
        json amb ok o error
- action = ListAllContainers
    parameters:
        String token        token de sessió
        int screen          pantalla des de la que es demana el llistat
        String filterField  columna per la que es vol filtrar. Si no és vàlida, ho mostra tot
        String filterValue  valor del camp que es vol filtrar. Si no és vàlida, ho mostra tot
    return:
        json amb un llistat dels contenidors, filtrats, de 10 en 10, situant el cursor segons el paràmetre screen, i informació de si està al començament o no i al final o no de la llista.
